```go
package bonusengine

import (
    "github.com/google/uuid"
    "github.com/shopspring/decimal"  // para numeric(14,2) sin pérdida
)

type PeriodID int64
type AffiliateID int64
type EventID int64

// Candidate represents one potential bonus payment for one ancestor
// triggered by one source event in the binary tree.
type Candidate struct {
    AffiliateID    AffiliateID
    SourceEventID  EventID
    AffiliatePackageID int64       // for package_cap enforcement

    Side           string          // "L" or "R"
    NewBlocks      int             // bloques nuevos a pagar
    PointsAfter    int             // min(L, R) tras este evento
    CarryAfter     int             // residuo

    GrossAmount    decimal.Decimal // blocks × bonus_per_block
    DailyReduction decimal.Decimal
    PackageReduction decimal.Decimal
}

// PlanConfig caches active plan_config row in memory during a period close
type PlanConfig struct {
    ID                int64
    BlockSize         int
    BonusPerBlock     decimal.Decimal
    DepthCap          int
    DailyCapFactor    decimal.Decimal
    LifetimeCapFactor decimal.Decimal
    TreasuryAlpha     decimal.Decimal
    CarryDecayDays    int
    QualifiedDirectsLeft  int
    QualifiedDirectsRight int
}

// PeriodSnapshot is the immutable result of closing a period
type PeriodSnapshot struct {
    PeriodID         PeriodID
    InflowsTotal     decimal.Decimal
    EventsCount      int
    CandidatesCount  int
    ProjectedOutflows decimal.Decimal
    Theta            decimal.Decimal
    TotalPaid        decimal.Decimal
}
```

## Algoritmo principal (`binary_close.go`)

```go
package bonusengine

import (
    "context"
    "errors"
    "fmt"
    "time"

    "github.com/jackc/pgx/v5"
    "github.com/shopspring/decimal"
)

var (
    ErrPeriodNotOpen      = errors.New("period is not in 'open' status")
    ErrConcurrentClose    = errors.New("another close is already in progress")
    ErrSolvencyBreach     = errors.New("calculated payouts breach treasury alpha")
    ErrNoActivePlanConfig = errors.New("no active plan_config")
)

// CloseBinaryPeriod ejecuta el cierre de un período binario.
//
// Es idempotente: re-ejecutar sobre un período ya cerrado retorna el snapshot
// existente sin duplicar pagos. La idempotencia se ancla en
// binary_block_payment.UNIQUE(affiliate_id, period_id, source_event_id, posted_at).
//
// Concurrencia: una sola transacción SERIALIZABLE per-período. Si dos llamadas
// concurrentes intentan cerrar el mismo período, la segunda aborta con
// ErrConcurrentClose (advisory lock por period_id).
//
// Implementa el algoritmo de mlm_binario_estabilidad.md §6.
func (e *Engine) CloseBinaryPeriod(ctx context.Context, periodID PeriodID) (*PeriodSnapshot, error) {
    log := e.log.With().Int64("period_id", int64(periodID)).Logger()
    timer := prometheus.NewTimer(e.closeRunDuration)
    defer timer.ObserveDuration()

    // ─────────────────────────────────────────────────────────────────────
    // 0. Idempotencia: si ya está closed, retornar snapshot existente
    // ─────────────────────────────────────────────────────────────────────
    period, err := e.queries.GetBinaryPeriod(ctx, int64(periodID))
    if err != nil {
        return nil, fmt.Errorf("load period: %w", err)
    }
    if period.Status == "closed" {
        log.Info().Msg("period already closed; returning existing snapshot")
        return periodToSnapshot(period), nil
    }
    if period.Status != "open" {
        return nil, fmt.Errorf("%w: status=%s", ErrPeriodNotOpen, period.Status)
    }

    // ─────────────────────────────────────────────────────────────────────
    // 1. Advisory lock para prevenir close concurrente del mismo período
    // ─────────────────────────────────────────────────────────────────────
    if err := e.queries.AcquireLock(ctx, int64(periodID)); err != nil {
        return nil, fmt.Errorf("%w: %v", ErrConcurrentClose, err)
    }
    defer e.queries.ReleaseLock(ctx, int64(periodID))

    // ─────────────────────────────────────────────────────────────────────
    // 2. Cargar plan_config vigente para este período
    // ─────────────────────────────────────────────────────────────────────
    plan, err := e.loadActivePlanConfig(ctx, period.PeriodStart)
    if err != nil {
        return nil, fmt.Errorf("load plan config: %w", err)
    }
    log.Info().Int64("plan_config_id", plan.ID).Msg("loaded active plan config")

    // ─────────────────────────────────────────────────────────────────────
    // 3. SERIALIZABLE transaction: el resto del cierre
    // ─────────────────────────────────────────────────────────────────────
    tx, err := e.db.BeginTx(ctx, pgx.TxOptions{IsoLevel: pgx.Serializable})
    if err != nil {
        return nil, fmt.Errorf("begin tx: %w", err)
    }
    defer tx.Rollback(ctx) // safe to call after Commit

    qtx := e.queries.WithTx(tx)

    // ─────────────────────────────────────────────────────────────────────
    // 4. Marcar period como closing
    // ─────────────────────────────────────────────────────────────────────
    if err := qtx.SetPeriodStatus(ctx, sqlc.SetPeriodStatusParams{
        ID:     int64(periodID),
        Status: "closing",
    }); err != nil {
        return nil, fmt.Errorf("set closing status: %w", err)
    }

    // ─────────────────────────────────────────────────────────────────────
    // 5. Congelar inflows del período
    // ─────────────────────────────────────────────────────────────────────
    inflows, err := qtx.SumInflowsForPeriod(ctx, sqlc.SumInflowsForPeriodParams{
        PeriodStart: period.PeriodStart,
        PeriodEnd:   period.PeriodEnd,
    })
    if err != nil {
        return nil, fmt.Errorf("sum inflows: %w", err)
    }
    log.Info().Str("inflows", inflows.String()).Msg("inflows frozen")

    // ─────────────────────────────────────────────────────────────────────
    // 6. Enumerar candidates (pagos potenciales)
    //    Para cada evento del período, recorrer ancestros hasta depth_cap,
    //    saltando los no-qualified.
    // ─────────────────────────────────────────────────────────────────────
    candidates, eventsCount, err := e.enumerateCandidates(ctx, qtx, period, plan)
    if err != nil {
        return nil, fmt.Errorf("enumerate candidates: %w", err)
    }
    e.candidatesCounter.Add(float64(len(candidates)))
    log.Info().
        Int("events", eventsCount).
        Int("candidates", len(candidates)).
        Msg("candidates enumerated")

    // ─────────────────────────────────────────────────────────────────────
    // 7. Calcular projected outflows + theta
    //    theta = clamp(alpha × inflows / projected, 0, 1)
    // ─────────────────────────────────────────────────────────────────────
    var projected decimal.Decimal
    for _, c := range candidates {
        projected = projected.Add(c.GrossAmount)
    }

    theta := computeTheta(plan.TreasuryAlpha, inflows, projected)
    e.thetaGauge.Set(theta.InexactFloat64())

    log.Info().
        Str("projected", projected.String()).
        Str("theta", theta.String()).
        Msg("theta computed")

    // ─────────────────────────────────────────────────────────────────────
    // 8. Aplicar pagos: para cada candidate, INSERT binary_block_payment +
    //    crear transaction + UPDATE binary_node_state + UPDATE package_cap_state.
    //    Los triggers en DB enforce T2 (package cap), T3 (daily cap).
    //    Si CUALQUIER trigger falla, toda la transacción aborta.
    // ─────────────────────────────────────────────────────────────────────
    var totalPaid decimal.Decimal
    nowTs := time.Now()

    for _, c := range candidates {
        netAmount := c.GrossAmount.Mul(theta).Round(2)
        if netAmount.IsZero() {
            continue
        }

        // 8.1 Crear transaction idempotente con external_ref único
        externalRef := fmt.Sprintf("binary:%d:%d:%d", periodID, c.AffiliateID, c.SourceEventID)
        txnID, err := qtx.UpsertTransaction(ctx, sqlc.UpsertTransactionParams{
            ExternalRef: externalRef,
            Description: fmt.Sprintf("binary bonus period=%d", periodID),
            Status:      "pending",
        })
        if err != nil {
            return nil, fmt.Errorf("upsert txn: %w", err)
        }

        // 8.2 Insert binary_block_payment (los triggers validan caps)
        _, err = qtx.InsertBlockPayment(ctx, sqlc.InsertBlockPaymentParams{
            BinaryPeriodID:      int64(periodID),
            PlanConfigID:        plan.ID,
            AffiliateID:         int64(c.AffiliateID),
            SourceEventID:       int64(c.SourceEventID),
            Blocks:              int32(c.NewBlocks),
            GrossAmount:         c.GrossAmount,
            ThetaApplied:        theta,
            NetAmount:           netAmount,
            CapDailyReduction:   c.DailyReduction,
            CapPackageReduction: c.PackageReduction,
            TransactionID:       txnID,
            PostedAt:            nowTs,
        })
        if err != nil {
            // Triggers de cap dispararon: log y continúa (este candidate se
            // recorta automáticamente — el net se ajusta antes de re-INSERT)
            // En la práctica los caps ya se aplicaron en GrossAmount al enumerar;
            // este branch atrapa breakage de race conditions.
            log.Warn().Err(err).
                Int64("affiliate_id", int64(c.AffiliateID)).
                Msg("cap trigger fired during INSERT; skipping candidate")
            continue
        }

        // 8.3 Crear el wallet_movement vía cliente al Ledger
        //     (en producción es una llamada interna al módulo ledger del mismo
        //     vp-engine; aquí simplificado)
        if err := e.postLedgerMovement(ctx, qtx, txnID, c, netAmount, nowTs); err != nil {
            return nil, fmt.Errorf("post ledger movement: %w", err)
        }

        // 8.4 Update binary_node_state (counters)
        if err := qtx.UpsertNodeState(ctx, sqlc.UpsertNodeStateParams{
            AffiliateID:      int64(c.AffiliateID),
            BinaryPeriodID:   int64(periodID),
            BlocksPaidLeft:   int32(blocksAdded(c, "L")),
            BlocksPaidRight:  int32(blocksAdded(c, "R")),
            CarryLeft:        int32(carryAfter(c, "L")),
            CarryRight:       int32(carryAfter(c, "R")),
            CarryStartedAt:   nowTs,
            PaidTodayAmount:  netAmount,
            PaidTodayDate:    nowTs.In(bogotaTZ()).Truncate(24 * time.Hour),
            PaidPeriodAmount: netAmount,
        }); err != nil {
            return nil, fmt.Errorf("update node state: %w", err)
        }

        // 8.5 Update package_cap_state (trigger T2 enforce el cap)
        if err := qtx.IncrementPackageCap(ctx, sqlc.IncrementPackageCapParams{
            AffiliatePackageID: c.AffiliatePackageID,
            Increment:          netAmount,
        }); err != nil {
            // Si el trigger T2 disparó: cap alcanzado, paquete cerrado.
            // No es error fatal — log y continue.
            if isCapBreachError(err) {
                log.Info().Int64("pkg", c.AffiliatePackageID).Msg("package cap reached; closing")
                continue
            }
            return nil, fmt.Errorf("increment package cap: %w", err)
        }

        totalPaid = totalPaid.Add(netAmount)
        e.payoutsTotalUSD.Add(netAmount.InexactFloat64())
    }

    // ─────────────────────────────────────────────────────────────────────
    // 9. Cerrar el período con snapshot inmutable
    // ─────────────────────────────────────────────────────────────────────
    snapshot := &PeriodSnapshot{
        PeriodID:          periodID,
        InflowsTotal:      inflows,
        EventsCount:       eventsCount,
        CandidatesCount:   len(candidates),
        ProjectedOutflows: projected,
        Theta:             theta,
        TotalPaid:         totalPaid,
    }
    if err := qtx.ClosePeriod(ctx, sqlc.ClosePeriodParams{
        ID:                int64(periodID),
        InflowsTotal:      inflows,
        EventsCount:       int32(eventsCount),
        CandidatesCount:   int32(len(candidates)),
        ProjectedOutflows: projected,
        Theta:             theta,
        TotalPaid:         totalPaid,
        ClosedAt:          time.Now(),
    }); err != nil {
        return nil, fmt.Errorf("close period: %w", err)
    }

    // ─────────────────────────────────────────────────────────────────────
    // 10. Caducar carry vencido (β días)
    // ─────────────────────────────────────────────────────────────────────
    cutoff := time.Now().AddDate(0, 0, -plan.CarryDecayDays)
    if err := qtx.ExpireCarry(ctx, cutoff); err != nil {
        return nil, fmt.Errorf("expire carry: %w", err)
    }

    // ─────────────────────────────────────────────────────────────────────
    // 11. Verificar T1 (no overspend) — paranoia post-commit check
    // ─────────────────────────────────────────────────────────────────────
    maxAllowed := plan.TreasuryAlpha.Mul(inflows)
    if totalPaid.GreaterThan(maxAllowed.Add(decimal.NewFromFloat(0.01))) {
        e.solvencyBreaches.Inc()
        // Esto NO debería pasar si theta se calculó correctamente.
        // Si pasa: rollback + alerta crítica + investigación.
        return nil, fmt.Errorf("%w: paid=%s > max_allowed=%s",
            ErrSolvencyBreach, totalPaid, maxAllowed)
    }

    // ─────────────────────────────────────────────────────────────────────
    // 12. COMMIT
    // ─────────────────────────────────────────────────────────────────────
    if err := tx.Commit(ctx); err != nil {
        return nil, fmt.Errorf("commit: %w", err)
    }

    // ─────────────────────────────────────────────────────────────────────
    // 13. Emitir evento NATS para notifications + reporting
    // ─────────────────────────────────────────────────────────────────────
    if err := e.publishCloseEvent(ctx, snapshot); err != nil {
        // No fail; el cierre ya commiteó. Log y métricas.
        log.Warn().Err(err).Msg("failed to publish close event")
    }

    log.Info().
        Str("inflows", inflows.String()).
        Str("total_paid", totalPaid.String()).
        Str("theta", theta.String()).
        Int("payments", len(candidates)).
        Msg("binary period closed successfully")

    return snapshot, nil
}

// computeTheta calculates the throttle factor for the period.
// theta = clamp(alpha × inflows / projected, 0, 1)
func computeTheta(alpha, inflows, projected decimal.Decimal) decimal.Decimal {
    if projected.IsZero() {
        return decimal.NewFromInt(1)
    }
    raw := alpha.Mul(inflows).Div(projected)
    one := decimal.NewFromInt(1)
    if raw.GreaterThan(one) {
        return one
    }
    if raw.LessThan(decimal.Zero) {
        return decimal.Zero
    }
    return raw.Round(6) // mirror DB precision
}
```

## Helpers (resumidos)

```go
// enumerateCandidates walks tree_event for the period, for each event walks
// ancestors up to depth_cap, applies qualification + cap reductions in-memory.
//
// Returns the list of candidates ready for theta application.
func (e *Engine) enumerateCandidates(ctx context.Context, qtx *sqlc.Queries,
    period sqlc.BinaryPeriod, plan PlanConfig) ([]Candidate, int, error) {

    events, err := qtx.GetTreeEventsForPeriod(ctx, period.ID)
    if err != nil {
        return nil, 0, err
    }

    candidates := make([]Candidate, 0, len(events)*plan.DepthCap)

    for _, evt := range events {
        // Ancestors via ltree path (already in DB)
        ancestors, err := qtx.GetAncestorsLimit(ctx, sqlc.GetAncestorsLimitParams{
            AffiliateID: evt.AffiliateID,
            DepthCap:    int32(plan.DepthCap),
        })
        if err != nil { return nil, 0, err }

        for _, anc := range ancestors {
            if !isQualified(anc, plan) {
                continue // compresión: salta a siguiente ancestro
            }

            side := determineSide(anc.Path, evt.AffiliateID) // L or R

            blocks, carry := computeBlocks(anc, evt, plan, side)
            if blocks == 0 { continue }

            gross := plan.BonusPerBlock.Mul(decimal.NewFromInt(int64(blocks)))

            // Apply caps in-memory (DB triggers do final enforcement)
            grossAfterDaily, dailyRed := applyDailyCap(anc, gross, plan)
            grossAfterPkg, pkgRed := applyPackageCap(evt.PackageID, grossAfterDaily)

            candidates = append(candidates, Candidate{
                AffiliateID:        anc.ID,
                SourceEventID:      evt.ID,
                AffiliatePackageID: evt.PackageID,
                Side:               side,
                NewBlocks:          blocks,
                CarryAfter:         carry,
                GrossAmount:        grossAfterPkg,
                DailyReduction:     dailyRed,
                PackageReduction:   pkgRed,
            })
        }
    }
    return candidates, len(events), nil
}

// isQualified mirrors the SQL definition in mlm_binario_estabilidad.md §3.3
func isQualified(a sqlc.Affiliate, plan PlanConfig) bool {
    return a.Status == "active" &&
           a.QualifiedDirectsLeft  >= int64(plan.QualifiedDirectsLeft) &&
           a.QualifiedDirectsRight >= int64(plan.QualifiedDirectsRight) &&
           a.HasActivePackage
}
```

### Shadow mode

Antes del cutover, correr `CloseBinaryPeriodShadow()` en paralelo al motor legacy:
- Lee mismos eventos.
- Calcula nueva fórmula.
- Escribe resultados a `mlm.binary_block_payment_shadow` (tabla idéntica, schema separado).
- Compara nightly: `SELECT * FROM v_shadow_diff` muestra divergencias.

Cero divergencias > 0.01 % por afiliado durante 30 días = OK para cutover.
